function doGet() {
  var cal = CalendarApp.getDefaultCalendar();
  var today = new Date();
  var start = new Date(today.getFullYear(),today.getMonth(),today.getDate());
  var end = new Date(today.getFullYear(),today.getMonth(),today.getDate()+1);
  var events = cal.getEvents(start, end);
  var data = events.map(function(e) {
    var st = e.getStartTime();
    var et = e.getEndTime();
    return {
      title: e.getTitle(),
      start: st.toISOString(),
      end: et.toISOString(),
      allDay: e.isAllDayEvent(),
      location: e.getLocation()
    };
  });
  var out = ContentService.createTextOutput(JSON.stringify({events:data}));
  out.setMimeType(ContentService.MimeType.JSON);
  return out;
}

校園安全 艾森豪矩陣 — 使用說明
整合情緒感知、AI 事件分類、校安法定通報程序、Google 行事曆的校長行政決策工具。
四象限定義（校長視角）
重要・緊急 → 立即處理
攸關人身安全、法規或重大校譽，必須馬上處理。
例：學生受傷、性侵通報、火災
重要・不緊急 → 排程規劃
對長遠發展有幫助，沒有立即壓迫感。
例：防災演練、霸凌調查、IEP 會議
緊急・不重要 → 委派速決
時間緊迫的例行庶務，對核心教育目標影響小。
例：臨時報表、設備故障、轉交物品
不重要・不緊急 → 刪除
干擾性事物，對校務運作無實質幫助。
例：廣告信、八卦謠言、冗長非必要會議
校安通報時限（依教育部校安通報要點）
🚨 緊急事件 — 知悉後 2 小時內電話通報
死傷、性侵知悉、兒虐知悉、媒體到場、逾越學校處理能力、天然災害
⏰ 依法規通報 — 知悉後 24 小時內校安網
霸凌、體罰、藥物濫用、法定傳染病、食物中毒、兒少保護
📋 一般校安 — 知悉後 72 小時內校安網
職場霸凌、一般安全事件、輕微校安通報
